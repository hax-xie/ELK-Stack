## Docker Compose 部署

### 1. 创建 docker-compose.yml
```yaml
# 将此文件保存为 docker-compose.yml
version: '3.8'
services:
  elasticsearch:
    image: docker.elastic.co/elasticsearch/elasticsearch:8.13.0
    container_name: elasticsearch
    environment:
      - discovery.type=single-node
      - ES_JAVA_OPTS=-Xms512m -Xmx512m
      - xpack.security.enabled=false
    ports:
      - "9200:9200"
      - "9300:9300"
    volumes:
      - es_data:/usr/share/elasticsearch/data
    ulimits:
      memlock:
        soft: -1
        hard: -1
    networks:
      - elk

  logstash:
    image: docker.elastic.co/logstash/logstash:8.13.0
    container_name: logstash
    volumes:
      - ./config/logstash/config:/usr/share/logstash/config
      - ./config/logstash/pipeline:/usr/share/logstash/pipeline
    ports:
      - "5044:5044"
      - "9600:9600"
    depends_on:
      - elasticsearch
    networks:
      - elk

  kibana:
    image: docker.elastic.co/kibana/kibana:8.13.0
    container_name: kibana
    environment:
      - ELASTICSEARCH_HOSTS=http://elasticsearch:9200
    ports:
      - "5601:5601"
    depends_on:
      - elasticsearch
    networks:
      - elk

volumes:
  es_data:

networks:
  elk:
    driver: bridge
```

### 2. 创建 Logstash 配置文件
```bash
# 创建配置目录
mkdir -p config/logstash/config
mkdir -p config/logstash/pipeline
```

#### **logstash.conf**
```conf
# config/logstash/pipeline/logstash.conf
input {
  beats {
    port => 5044
  }
  file {
    path => "/var/log/syslog"
    type => "system"
    start_position => "beginning"
  }
  file {
    path => "/var/log/auth.log"
    type => "auth"
    start_position => "beginning"
  }
}

filter {
  # 系统日志解析
  if [type] == "system" {
    grok {
      match => { "message" => "%{TIMESTAMP_ISO8601:timestamp} %{LOGLEVEL:level} %{GREEDYDATA:message}" }
    }
    date {
      match => ["timestamp", "ISO8601"]
      target => "@timestamp"
    }
  }

  # 认证日志解析
  if [type] == "auth" {
    grok {
      match => { "message" => "%{TIMESTAMP_ISO8601:timestamp} %{GREEDYDATA:message}" }
    }
    date {
      match => ["timestamp", "ISO8601"]
      target => "@timestamp"
    }
  }
  
  # 统一字段处理
  mutate {
    remove_field => ["host"]
    add_field => {
      "environment" => "production"
    }
  }
}

output {
  elasticsearch {
    hosts => ["http://elasticsearch:9200"]
    index => "system-logs-%{+YYYY.MM.dd}"
  }
  stdout {
    codec => rubydebug
  }
}
```

#### **logstash.yml**
```yaml
# config/logstash/config/logstash.yml
http.host: "0.0.0.0"
http.port: 9600
log.level: info
queue.type: memory
path.data: /usr/share/logstash/data
```

## 部署步骤

### 步骤1：下载配置文件
```bash
# 保存所有配置文件到项目目录
# docker-compose.yml
# config/logstash/config/logstash.yml
# config/logstash/pipeline/logstash.conf
```

### 步骤2：启动ELK服务
```bash
# 启动ELK服务
docker-compose up -d

# 查看服务状态
docker-compose ps

# 检查日志
docker-compose logs elasticsearch
docker-compose logs logstash
docker-compose logs kibana
```

### 步骤3：验证服务
```bash
# 验证Elasticsearch
curl -X GET "http://localhost:9200/_cluster/health?pretty"

# 验证Kibana（等待几分钟后）
curl -X GET "http://localhost:5601/api/status" -I

# 验证Logstash
curl -X GET "http://localhost:9600/?pretty"
```

### 步骤4：配置Filebeat收集器
```bash
# 创建Filebeat配置文件
cat > filebeat.yml << 'EOF'
filebeat.inputs:
- type: log
  enabled: true
  paths:
    - /var/log/syslog
    - /var/log/auth.log
    - /var/log/app/*.log
  
  fields:
    log_type: system
    environment: production
  
  processors:
    - add_host_metadata:
        netinfo.enabled: false
    - add_cloud_metadata: ~

output.logstash:
  hosts: ["localhost:5044"]
EOF
```

## 常见问题解决

### 1. Elasticsearch启动失败
```bash
# 检查内存设置
# 修改 docker-compose.yml 中的 ES_JAVA_OPTS
# 增加内存：ES_JAVA_OPTS=-Xms1g -Xmx1g

# 检查权限问题
chmod 777 ./config -R
```

### 2. Logstash连接失败
```bash
# 检查网络连接
docker network inspect elk

# 检查端口是否开放
netstat -tlnp | grep 5044
```

### 3. Kibana无法访问
```bash
# 等待Kibana启动完成（可能需要几分钟）
# 检查日志
docker logs kibana

# 验证Elasticsearch连接
curl -X GET "http://elasticsearch:9200"
```

## 安全配置（可选）

### 1. 启用基本安全
```yaml
# 修改 elasticsearch 环境变量
environment:
  - discovery.type=single-node
  - ES_JAVA_OPTS=-Xms512m -Xmx512m
  - xpack.security.enabled=true
  - xpack.security.transport.ssl.enabled=true
```

### 2. 设置密码
```bash
# 进入Elasticsearch容器设置密码
docker exec -it elasticsearch elasticsearch-setup-passwords auto

# 配置Kibana连接
# 修改 kibana 环境变量
environment:
  - ELASTICSEARCH_HOSTS=http://elasticsearch:9200
  - ELASTICSEARCH_USERNAME=kibana_user
  - ELASTICSEARCH_PASSWORD=kibana_password
```
